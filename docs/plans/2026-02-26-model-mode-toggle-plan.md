# Model Mode Toggle Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a UI toggle to switch all agents between a cheap testing model (amazon/nova-micro-v1) and their production models.

**Architecture:** A shadcn Switch in the nav bar sends `modelMode` in the request body. The workflow stores it in state, each step puts it on RequestContext, and each agent's dynamic `model` function reads it to select the model. localStorage persists the user's choice.

**Tech Stack:** shadcn Switch, React hooks, Mastra RequestContext, Zod schema extension.

### Task 6: Create useModelMode hook

**Files:**

- Create: `src/hooks/use-model-mode.ts`

**Step 1: Create the hook**

```typescript
'use client';

import { useCallback, useSyncExternalStore } from 'react';

type ModelMode = 'testing' | 'production';

const STORAGE_KEY = 'lo-solver-model-mode';
const DEFAULT_MODE: ModelMode = 'testing';

function getSnapshot(): ModelMode {
  if (typeof window === 'undefined') return DEFAULT_MODE;
  const stored = localStorage.getItem(STORAGE_KEY);
  return stored === 'production' ? 'production' : 'testing';
}

function getServerSnapshot(): ModelMode {
  return DEFAULT_MODE;
}

function subscribe(callback: () => void): () => void {
  window.addEventListener('storage', callback);
  return () => window.removeEventListener('storage', callback);
}

export function useModelMode(): [ModelMode, (mode: ModelMode) => void] {
  const mode = useSyncExternalStore(subscribe, getSnapshot, getServerSnapshot);

  const setMode = useCallback((newMode: ModelMode) => {
    localStorage.setItem(STORAGE_KEY, newMode);
    // Trigger re-render in this tab (storage event only fires cross-tab)
    window.dispatchEvent(new StorageEvent('storage', { key: STORAGE_KEY }));
  }, []);

  return [mode, setMode];
}
```

Uses `useSyncExternalStore` for SSR-safe localStorage access without hydration mismatches.

import { Switch } from '@/components/ui/switch';
import { useModelMode } from '@/hooks/use-model-mode';

export function ModelModeToggle() {
  const [mode, setMode] = useModelMode();

  return (
    <div className="flex items-center gap-2 text-sm">
      <span className={mode === 'testing' ? 'text-foreground' : 'text-muted-foreground'}>
        Testing ($)
      </span>
      <Switch
        checked={mode === 'production'}
        onCheckedChange={(checked) => setMode(checked ? 'production' : 'testing')}
      />
      <span className={mode === 'production' ? 'text-foreground' : 'text-muted-foreground'}>
        Production ($$$)
      </span>
    </div>
  );
}
