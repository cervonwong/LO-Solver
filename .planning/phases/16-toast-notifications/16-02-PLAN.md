   ```typescript
   /**
    * Extract cost from an agent response, accumulate in RequestContext,
    * and emit a data-cost-update event if a $1 boundary is crossed.
    *
    * Works with both generateWithRetry (Agent.generate result) and
    * streamWithRetry (FullOutput) responses.
    */
   export async function trackCost(
     requestContext: RequestContextGetter,
     writer: StepWriter | undefined,
     result: { steps?: Array<{ providerMetadata?: Record<string, unknown> }>; providerMetadata?: Record<string, unknown> },
   ): Promise<void> {
     if (!requestContext) return;

     // Sum cost across all steps (multi-step agents) or use top-level (single-step)
     let callCost = 0;
     if (result.steps && result.steps.length > 0) {
       for (const step of result.steps) {
         const stepCost = (step.providerMetadata as any)?.openrouter?.usage?.cost;
         if (typeof stepCost === 'number') callCost += stepCost;
       }
     }
     if (callCost === 0) {
       // Fallback to top-level providerMetadata (single-step or non-step result)
       const topCost = (result.providerMetadata as any)?.openrouter?.usage?.cost;
       if (typeof topCost === 'number') callCost = topCost;
     }

     if (callCost <= 0) return;

     const prevCost = (requestContext.get('cumulative-cost') as number) ?? 0;
     const newCost = prevCost + callCost;

     // We cannot call requestContext.set() from the getter interface,
     // so we need to track cost via a mutable object pattern.
     // Instead, use a wrapper that stores cost in the context.
     // NOTE: The callers (step files) have the full requestContext with set().
     // This helper will be called with the result and the step will handle
     // the context update. Return the new cost and whether a threshold crossed.

     // Actually, since RequestContextGetter is read-only, restructure:
     // Return the cost info and let the caller handle context update + emission.
   }
   ```

   **IMPORTANT CORRECTION**: The `RequestContextGetter` used in helpers is read-only (`get` only). The workflow step files have the full `RequestContext` with `set()`. So the helper should be a pure function that computes cost and threshold crossing, and the step files handle the mutation and emission.

   Instead, create two helpers:
   ```typescript
   /**
    * Extract total cost from an agent response.
    * Sums per-step costs for multi-step agents, falls back to top-level cost.
    */
   export function extractCostFromResult(
     result: { steps?: Array<{ providerMetadata?: Record<string, unknown> }>; providerMetadata?: Record<string, unknown> },
   ): number {
     let callCost = 0;
     if (result.steps && result.steps.length > 0) {
       for (const step of result.steps) {
         const stepCost = (step.providerMetadata as any)?.openrouter?.usage?.cost;
         if (typeof stepCost === 'number') callCost += stepCost;
       }
     }
     if (callCost === 0) {
       const topCost = (result.providerMetadata as any)?.openrouter?.usage?.cost;
       if (typeof topCost === 'number') callCost = topCost;
     }
     return callCost;
   }

   /**
    * Update cumulative cost in RequestContext and emit data-cost-update events
    * at each $1 boundary crossing.
    * Must be called with the full RequestContext (has set()) from step files.
    */
   export async function updateCumulativeCost(
     requestContext: { get: (key: keyof WorkflowRequestContext) => unknown; set: (key: keyof WorkflowRequestContext, value: unknown) => void },
     writer: StepWriter | undefined,
     callCost: number,
   ): Promise<void> {
     if (callCost <= 0) return;

     const prevCost = (requestContext.get('cumulative-cost') as number) ?? 0;
     const newCost = prevCost + callCost;
     requestContext.set('cumulative-cost', newCost);

     // Check if we crossed a $1 boundary
     const prevBucket = Math.floor(prevCost);
     const newBucket = Math.floor(newCost);
     if (newBucket > prevBucket) {
       // Emit one event for each $1 boundary crossed (handles >$1 single calls)
       for (let i = prevBucket + 1; i <= newBucket; i++) {
         await emitTraceEvent(writer, {
           type: 'data-cost-update',
           data: {
             cumulativeCost: i,
             timestamp: new Date().toISOString(),
           },
         });
       }
     }
   }
   ```
