3. **Remove the fallback block in the `data-tool-call` handler.** Currently (lines 291-296):
   ```typescript
   // Fallback: assign to the most recently opened active agent
   if (activeAgentStack.length > 0) {
     const fallbackAgent = activeAgentStack[activeAgentStack.length - 1]!;
     fallbackAgent.toolCalls.push(event as ToolCallEvent);
     fallbackAgent.children.push(event as ToolCallEvent);
     continue;
   }
   ```
   Replace this entire block (including the TODO comment above it) with a `console.warn` that logs the orphan:
   ```typescript
   // Orphaned tool call — no matching parent agent found
   console.warn(
     `[trace] Orphaned tool call: toolName=${event.data.toolName}, parentId=${'parentId' in event.data ? event.data.parentId : 'missing'}`,
   );
   ```
   Then let execution fall through to the standalone `result.push(event)` at the bottom of the loop. Do NOT `continue` — the orphaned event will be pushed to `result` as a standalone event at the root level.

4. **Clean up the data-tool-call handler** so the logic flow is clear:
   ```typescript
   if (event.type === 'data-tool-call') {
     // Try explicit parentId
     if ('parentId' in event.data && event.data.parentId) {
       const parentGroup = agentMap.get(event.data.parentId);
       if (parentGroup) {
         parentGroup.toolCalls.push(event as ToolCallEvent);
         parentGroup.children.push(event as ToolCallEvent);
         continue;
       }
     }
     // Orphaned tool call — no matching parent agent found
     console.warn(
       `[trace] Orphaned tool call: toolName=${event.data.toolName}, parentId=${'parentId' in event.data ? event.data.parentId : 'missing'}`,
     );
     // Falls through to result.push(event) below
   }
   ```
