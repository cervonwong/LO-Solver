**B. Create `src/app/api/evals/[id]/route.ts`:**

Follow the exact pattern from `src/app/api/examples/[id]/route.ts` for the params signature (Next.js 15+ uses `Promise<{ id: string }>`).

```typescript
import { NextResponse } from 'next/server';
import { loadEvalRun } from '@/evals/storage';

export async function GET(
  request: Request,
  { params }: { params: Promise<{ id: string }> },
) {
  const { id } = await params;
  const run = await loadEvalRun(id);
  if (!run) {
    return NextResponse.json({ error: 'Not found' }, { status: 404 });
  }
  return NextResponse.json(run);
}
```
