**src/app/api/solve/route.ts:**
1. Import `claudeCode` from `@/mastra/claude-code-provider`
2. Import `generateText` from `'ai'` (lightweight, no structured output needed for auth probe)
3. After extracting `params.inputData`, read `providerMode` from `params.inputData?.providerMode`
4. Keep existing API key check but scope it: only require API key when providerMode is NOT 'claude-code':
   ```typescript
   const providerMode = params.inputData?.providerMode ?? 'openrouter-testing';
   if (providerMode !== 'claude-code' && !apiKey && !process.env.OPENROUTER_API_KEY) {
     return new Response(JSON.stringify({ error: 'No API key provided' }), { status: 401, ... });
   }
   ```
5. Add Claude Code auth gate when providerMode is 'claude-code':
   ```typescript
   if (providerMode === 'claude-code') {
     try {
       await generateText({
         model: claudeCode('claude-sonnet-4-6'),
         prompt: 'Respond with OK',
         maxTokens: 10,
       });
     } catch (error) {
       const message = error instanceof Error ? error.message : String(error);
       if (message.includes('authentication_failed') || message.includes('not authenticated') || message.includes('spawn') || message.includes('ENOENT')) {
         return new Response(
           JSON.stringify({
             error: 'Claude Code is not authenticated. Run `claude login` in your terminal, then try again.',
           }),
           { status: 401, headers: { 'Content-Type': 'application/json' } },
         );
       }
       // Re-throw non-auth errors
       return new Response(
         JSON.stringify({ error: `Claude Code error: ${message}` }),
         { status: 500, headers: { 'Content-Type': 'application/json' } },
       );
     }
   }
   ```
   The auth probe uses `generateText` from 'ai' SDK (not `query()` from agent SDK directly) because it goes through the same provider path as actual agent calls, catching both CLI-not-installed and auth-failed scenarios.
