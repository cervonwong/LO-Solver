```typescript
export async function emitToolTraceEvent(
  requestContext: RequestContextGetter,
  event: { type: string; data: Record<string, unknown> },
): Promise<void> {
  if (!requestContext) return;
  const writer = requestContext.get('step-writer') as StepWriter | undefined;
  const parentAgentId = requestContext.get('parent-agent-id') as string | undefined;
  // Inject parentId and id into the event data if it's a tool-call event
  const enrichedData = event.type === 'data-tool-call'
    ? { ...event.data, id: generateEventId(), parentId: parentAgentId ?? '' }
    : event.data;
  await emitTraceEvent(writer, { type: event.type, data: enrichedData });
}
```

  <action>
Replace all `generateWithRetry` calls in workflow.ts with `streamWithRetry`. For each agent call, wrap it with agent-start/agent-end events and set parent-agent-id in the request context.

**Pattern to apply to every agent call** (there are 11 calls — apply systematically):

Before (current pattern):
```typescript
const response = await generateWithRetry(
  mastra.getAgentById('agent-name'),
  { prompt, options: { maxSteps: 100, requestContext, structuredOutput: { schema } } },
);
// ... emit AgentReasoningEvent ...
```

After (new pattern):
```typescript
const agentEventId = generateEventId();
await emitTraceEvent(writer, {
  type: 'data-agent-start',
  data: {
    id: agentEventId,
    stepId,
    agentName: 'Agent Display Name',
    model: activeModelId(modelMode, 'model/name'),
    task: prompt.slice(0, 500),  // Truncate task for event size
    timestamp: new Date().toISOString(),
  },
});

// Set parent-agent-id so tools link their events to this agent
requestContext.set('parent-agent-id', agentEventId);

const startTime = new Date();
const response = await streamWithRetry(
  mastra.getAgentById('agent-name'),
  {
    prompt,
    options: { maxSteps: 100, requestContext, structuredOutput: { schema } },
    onTextChunk: (chunk) => {
      // Fire-and-forget text chunk event (don't await in hot path)
      emitTraceEvent(writer, {
        type: 'data-agent-text-chunk',
        data: { parentId: agentEventId, text: chunk, timestamp: new Date().toISOString() },
      });
    },
  },
);
const durationMs = new Date().getTime() - startTime.getTime();

await emitTraceEvent(writer, {
  type: 'data-agent-end',
  data: {
    id: agentEventId,
    stepId,
    agentName: 'Agent Display Name',
    reasoning: response.text || '',
    durationMs,
    attempt: 1,     // streamWithRetry handles retries internally
    totalAttempts: 1,
    timestamp: new Date().toISOString(),
  },
});

// Clear parent-agent-id after agent completes
requestContext.set('parent-agent-id', undefined);
```

**Important notes for the conversion:**

1. `streamWithRetry` returns `FullOutput` which has `.text` (string, not Promise) and `.object` (the structured output). The current code accesses `response.text` and `response.object` — these remain the same on FullOutput.

2. For `response.reasoning` (used in `logAgentOutput` calls), FullOutput has `.reasoningText` (string | undefined). Update the logging calls accordingly: `response.reasoningText` instead of `response.reasoning`.

3. Remove ALL the old `emitTraceEvent` calls that emit `data-agent-reasoning` — these are replaced by the agent-start/end pair above.

4. For agent calls inside `hypothesizePromises` (parallel hypothesizers) and `verifyPromises` (parallel verifiers), the parent-agent-id should be set on each perspective's own `perspectiveRequestContext` or `verifyRequestContext`, not the shared `mainRequestContext`, to avoid race conditions.

5. For the convergence check calls that use `convergenceRequestContext`, set parent-agent-id on that context.

6. Remove the `import { generateWithRetry }` and add `import { streamWithRetry }` from `./agent-utils`. Keep `generateWithRetry` import ONLY if still needed (check — the tool files import it independently, so workflow.ts may not need it).

7. Import `generateEventId` from `@/lib/workflow-events`.

8. The `recordStepTiming` calls remain unchanged — they're for logging, not events.

**Retry tracking:** `streamWithRetry` handles retries internally. To track attempt count, add an optional return field to `streamWithRetry`'s result or use a simpler approach: for now, set `attempt: 1, totalAttempts: 1` in agent-end events. A future enhancement can expose retry count from streamWithRetry via a wrapper object. The retry popover UI (Phase 7 or later) will consume these fields.
